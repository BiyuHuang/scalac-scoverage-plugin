83   private def formatLine(prevChars: List[Char], line: List[Char], offset: Int, isCovered: Boolean, blocks: List[CoveredBlock], acc: NodeSeq): NodeSeq = {
84     blocks match {
85       case Nil => acc ++ formatLinePart(prevChars ::: line, isCovered)
86       case block :: tail => {
87         val (currChars, nextChars) = line.splitAt(block.offset - offset)
88         if (isCovered == block.count > 0) {
89           formatLine(prevChars ::: currChars, nextChars, block.offset, isCovered, tail, acc)
90         } else {
91           val part = formatLinePart(prevChars ::: currChars, isCovered)
92           formatLine(List[Char](), nextChars, block.offset, block.count > 0, tail, acc ++ part)
93         }
94       }
95     }
96   }
