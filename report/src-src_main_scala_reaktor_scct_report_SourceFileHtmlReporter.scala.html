86     blocks match {
87       case Nil => acc ++ formatLinePart(prevChars ::: line, isCovered)
88       case block :: tail => {
89         val (currChars, nextChars) = line.splitAt(block.offset - offset)

92         } else {

94           formatLine(List[Char](), nextChars, block.offset, block.count > 0, tail, acc ++ part)
95         }
